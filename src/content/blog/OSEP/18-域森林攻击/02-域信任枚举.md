# 域信任枚举技术详解

## 1. 域信任基础

### 什么是域信任
```
域信任是Active Directory中域之间的关系:
├── 允许一个域的用户访问另一个域的资源
├── 信任可以是单向或双向
├── 信任可以是传递或非传递
└── 是跨域攻击的基础
```

### 信任类型
| 类型 | 方向 | 传递性 | 创建方式 |
|------|------|--------|----------|
| 父子信任 | 双向 | 传递 | 自动 |
| 树根信任 | 双向 | 传递 | 自动 |
| 快捷信任 | 双向 | 传递 | 手动 |
| 外部信任 | 单向/双向 | 非传递 | 手动 |
| 森林信任 | 单向/双向 | 传递 | 手动 |

## 2. 使用.NET枚举域信任

### 基本枚举代码
```csharp
using System;
using System.DirectoryServices.ActiveDirectory;

class EnumTrusts
{
    static void Main()
    {
        // 获取当前域
        Domain currentDomain = Domain.GetCurrentDomain();
        Console.WriteLine("Current Domain: " + currentDomain.Name);

        // 枚举所有信任关系
        TrustRelationshipInformationCollection trusts = currentDomain.GetAllTrustRelationships();

        foreach (TrustRelationshipInformation trust in trusts)
        {
            Console.WriteLine("\n--- Trust Information ---");
            Console.WriteLine("Source: " + trust.SourceName);
            Console.WriteLine("Target: " + trust.TargetName);
            Console.WriteLine("Trust Type: " + trust.TrustType);
            Console.WriteLine("Trust Direction: " + trust.TrustDirection);
        }
    }
}
```

## 3. 使用Win32 API枚举域信任

### DsEnumerateDomainTrusts API
```csharp
using System;
using System.Runtime.InteropServices;

class Win32TrustEnum
{
    [DllImport("Netapi32.dll", CharSet = CharSet.Auto, SetLastError = true)]
    static extern int DsEnumerateDomainTrusts(
        string ServerName,
        uint Flags,
        out IntPtr Domains,
        out uint DomainCount
    );

    [DllImport("Netapi32.dll")]
    static extern int NetApiBufferFree(IntPtr Buffer);

    [StructLayout(LayoutKind.Sequential, CharSet = CharSet.Auto)]
    struct DS_DOMAIN_TRUSTS
    {
        [MarshalAs(UnmanagedType.LPWStr)]
        public string NetbiosDomainName;
        [MarshalAs(UnmanagedType.LPWStr)]
        public string DnsDomainName;
        public uint Flags;
        public uint ParentIndex;
        public uint TrustType;
        public uint TrustAttributes;
        public IntPtr DomainSid;
        public Guid DomainGuid;
    }

    // 信任标志
    const uint DS_DOMAIN_IN_FOREST = 0x0001;
    const uint DS_DOMAIN_DIRECT_OUTBOUND = 0x0002;
    const uint DS_DOMAIN_TREE_ROOT = 0x0004;
    const uint DS_DOMAIN_PRIMARY = 0x0008;
    const uint DS_DOMAIN_NATIVE_MODE = 0x0010;
    const uint DS_DOMAIN_DIRECT_INBOUND = 0x0020;

    static void Main()
    {
        IntPtr domains;
        uint domainCount;

        // 枚举所有信任
        int result = DsEnumerateDomainTrusts(
            null,
            0x3F, // 所有标志
            out domains,
            out domainCount
        );

        if (result == 0)
        {
            Console.WriteLine("Found " + domainCount + " trusted domains:");

            IntPtr current = domains;
            int structSize = Marshal.SizeOf(typeof(DS_DOMAIN_TRUSTS));

            for (int i = 0; i < domainCount; i++)
            {
                DS_DOMAIN_TRUSTS trust = (DS_DOMAIN_TRUSTS)Marshal.PtrToStructure(
                    current, typeof(DS_DOMAIN_TRUSTS));

                Console.WriteLine("\n--- Domain Trust ---");
                Console.WriteLine("NetBIOS Name: " + trust.NetbiosDomainName);
                Console.WriteLine("DNS Name: " + trust.DnsDomainName);
                Console.WriteLine("Trust Type: " + trust.TrustType);
                Console.WriteLine("Flags: 0x" + trust.Flags.ToString("X"));

                // 解析标志
                if ((trust.Flags & DS_DOMAIN_IN_FOREST) != 0)
                    Console.WriteLine("  - In Forest");
                if ((trust.Flags & DS_DOMAIN_DIRECT_OUTBOUND) != 0)
                    Console.WriteLine("  - Direct Outbound Trust");
                if ((trust.Flags & DS_DOMAIN_DIRECT_INBOUND) != 0)
                    Console.WriteLine("  - Direct Inbound Trust");
                if ((trust.Flags & DS_DOMAIN_TREE_ROOT) != 0)
                    Console.WriteLine("  - Tree Root");
                if ((trust.Flags & DS_DOMAIN_PRIMARY) != 0)
                    Console.WriteLine("  - Primary Domain");

                current = IntPtr.Add(current, structSize);
            }

            NetApiBufferFree(domains);
        }
        else
        {
            Console.WriteLine("Error: " + result);
        }
    }
}
```

## 4. 使用LDAP枚举域信任

### LDAP查询信任对象
```csharp
using System;
using System.DirectoryServices;

class LdapTrustEnum
{
    static void Main()
    {
        // 连接到域
        DirectoryEntry entry = new DirectoryEntry("LDAP://DC=corp,DC=com");
        DirectorySearcher searcher = new DirectorySearcher(entry);

        // 搜索信任对象
        searcher.Filter = "(objectClass=trustedDomain)";
        searcher.PropertiesToLoad.Add("name");
        searcher.PropertiesToLoad.Add("trustDirection");
        searcher.PropertiesToLoad.Add("trustType");
        searcher.PropertiesToLoad.Add("trustAttributes");
        searcher.PropertiesToLoad.Add("trustPartner");

        foreach (SearchResult result in searcher.FindAll())
        {
            Console.WriteLine("\n--- Trusted Domain ---");
            Console.WriteLine("Name: " + result.Properties["name"][0]);
            Console.WriteLine("Trust Partner: " + result.Properties["trustPartner"][0]);

            // 解析信任方向
            int direction = (int)result.Properties["trustDirection"][0];
            string directionStr = "";
            switch (direction)
            {
                case 0: directionStr = "Disabled"; break;
                case 1: directionStr = "Inbound"; break;
                case 2: directionStr = "Outbound"; break;
                case 3: directionStr = "Bidirectional"; break;
            }
            Console.WriteLine("Trust Direction: " + directionStr);

            // 解析信任类型
            int trustType = (int)result.Properties["trustType"][0];
            string typeStr = "";
            switch (trustType)
            {
                case 1: typeStr = "Downlevel (Windows NT)"; break;
                case 2: typeStr = "Uplevel (Windows 2000+)"; break;
                case 3: typeStr = "MIT (Kerberos)"; break;
                case 4: typeStr = "DCE"; break;
            }
            Console.WriteLine("Trust Type: " + typeStr);

            // 解析信任属性
            int attributes = (int)result.Properties["trustAttributes"][0];
            Console.WriteLine("Trust Attributes: 0x" + attributes.ToString("X"));
            if ((attributes & 0x1) != 0) Console.WriteLine("  - Non-Transitive");
            if ((attributes & 0x2) != 0) Console.WriteLine("  - Uplevel Only");
            if ((attributes & 0x4) != 0) Console.WriteLine("  - Quarantined (SID Filtering)");
            if ((attributes & 0x8) != 0) Console.WriteLine("  - Forest Transitive");
            if ((attributes & 0x10) != 0) Console.WriteLine("  - Cross Organization");
            if ((attributes & 0x20) != 0) Console.WriteLine("  - Within Forest");
            if ((attributes & 0x40) != 0) Console.WriteLine("  - Treat as External");
        }
    }
}
```

```powershell

Import-Module .\PowerView.ps1

# 枚举域信任
Get-DomainTrust

Get-DomainUser -Domain external.domain

# 枚举外部域组
Get-DomainGroup -Domain external.domain

# 枚举外部域计算机
Get-DomainComputer -Domain external.domain

# 查找外部域组成员
Get-DomainForeignGroupMember -Domain external.domain
```

### 使用AD模块
```powershell
# 导入AD模块
Import-Module ActiveDirectory

# 枚举信任
Get-ADTrust -Filter *

# 获取详细信息
Get-ADTrust -Filter * | Select-Object Name, Direction, TrustType, TrustAttributes

# 枚举特定域的信任
Get-ADTrust -Filter * -Server dc.external.domain
```

### 使用nltest
```cmd
# 枚举域信任
nltest /domain_trusts

# 枚举所有信任
nltest /domain_trusts /all_trusts

# 枚举特定域的信任
nltest /domain_trusts /v

# 获取域控制器
nltest /dclist:corp.com
```

## 6. 跨域枚举

### 枚举外部域对象
```csharp
using System;
using System.DirectoryServices;

class CrossDomainEnum
{
    static void Main()
    {
        // 连接到外部域
        string externalDomain = "LDAP://DC=external,DC=com";
        DirectoryEntry entry = new DirectoryEntry(externalDomain);
        DirectorySearcher searcher = new DirectorySearcher(entry);

        // 枚举用户
        Console.WriteLine("=== Users in External Domain ===");
        searcher.Filter = "(objectClass=user)";
        foreach (SearchResult result in searcher.FindAll())
        {
            Console.WriteLine(result.Properties["samAccountName"][0]);
        }

        // 枚举组
        Console.WriteLine("\n=== Groups in External Domain ===");
        searcher.Filter = "(objectClass=group)";
        foreach (SearchResult result in searcher.FindAll())
        {
            Console.WriteLine(result.Properties["name"][0]);
        }

        // 枚举计算机
        Console.WriteLine("\n=== Computers in External Domain ===");
        searcher.Filter = "(objectClass=computer)";
        foreach (SearchResult result in searcher.FindAll())
        {
            Console.WriteLine(result.Properties["name"][0]);
        }
    }
}
```

### 查找跨域组成员
```csharp
using System;
using System.DirectoryServices;

class CrossDomainGroupMembers
{
    static void Main()
    {
        // 查找包含外部域成员的组
        DirectoryEntry entry = new DirectoryEntry("LDAP://DC=corp,DC=com");
        DirectorySearcher searcher = new DirectorySearcher(entry);

        searcher.Filter = "(objectClass=group)";
        searcher.PropertiesToLoad.Add("name");
        searcher.PropertiesToLoad.Add("member");

        foreach (SearchResult result in searcher.FindAll())
        {
            if (result.Properties["member"].Count > 0)
            {
                foreach (string member in result.Properties["member"])
                {
                    // 检查是否是外部域成员
                    if (member.Contains("DC=external"))
                    {
                        Console.WriteLine("Group: " + result.Properties["name"][0]);
                        Console.WriteLine("  External Member: " + member);
                    }
                }
            }
        }
    }
}
```

## 7. 信任攻击准备

### 收集信任信息
```powershell
# 完整信任枚举脚本
function Invoke-TrustEnum {
    Write-Host "=== Domain Trust Enumeration ===" -ForegroundColor Green

    # 当前域信息
    $currentDomain = [System.DirectoryServices.ActiveDirectory.Domain]::GetCurrentDomain()
    Write-Host "`nCurrent Domain: $($currentDomain.Name)"
    Write-Host "Domain SID: $((Get-ADDomain).DomainSID)"

    # 枚举信任
    Write-Host "`n=== Trust Relationships ===" -ForegroundColor Yellow
    $trusts = $currentDomain.GetAllTrustRelationships()
    foreach ($trust in $trusts) {
        Write-Host "`nTarget: $($trust.TargetName)"
        Write-Host "  Type: $($trust.TrustType)"
        Write-Host "  Direction: $($trust.TrustDirection)"

        # 获取目标域SID
        try {
            $targetDomain = Get-ADDomain -Server $trust.TargetName
            Write-Host "  Target SID: $($targetDomain.DomainSID)"
        } catch {
            Write-Host "  Target SID: Unable to retrieve"
        }
    }

    # 森林信息
    Write-Host "`n=== Forest Information ===" -ForegroundColor Yellow
    $forest = [System.DirectoryServices.ActiveDirectory.Forest]::GetCurrentForest()
    Write-Host "Forest: $($forest.Name)"
    Write-Host "Forest Root Domain: $($forest.RootDomain)"

    # 森林中的域
    Write-Host "`nDomains in Forest:"
    foreach ($domain in $forest.Domains) {
        Write-Host "  - $($domain.Name)"
    }

    # 森林信任
    Write-Host "`n=== Forest Trusts ===" -ForegroundColor Yellow
    $forestTrusts = $forest.GetAllTrustRelationships()
    foreach ($trust in $forestTrusts) {
        Write-Host "`nTarget Forest: $($trust.TargetName)"
        Write-Host "  Type: $($trust.TrustType)"
        Write-Host "  Direction: $($trust.TrustDirection)"
    }
}

Invoke-TrustEnum
```
